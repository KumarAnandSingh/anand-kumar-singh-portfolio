## 3. Modern News Backend API

### **Scalable Architecture**
Built for high performance and reliability with modern backend practices:

#### **API Design & Implementation**
```javascript
// Express.js API with comprehensive middleware
const express = require('express');
const app = express();

// Security middleware
app.use(helmet());
app.use(cors(corsOptions));
app.use(rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutes
  max: 100 // requests per window
}));

// Article routes with caching
app.get('/api/articles', cacheMiddleware(300), async (req, res) => {
  const { page, category, search } = req.query;
  
  try {
    const articles = await Article.findPaginated({
      page: parseInt(page) || 1,
      category,
      search,
      status: 'published'
    });
    
    res.json({
      articles: articles.data,
      pagination: articles.meta,
      cache: true
    });
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch articles' });
  }
});

// Real-time updates via WebSocket
const io = require('socket.io')(server);
io.on('connection', (socket) => {
  socket.on('subscribe-updates', (userId) => {
    socket.join(`user-${userId}`);
  });
});
```

#### **Database Architecture**
```sql
-- Optimized database schema
CREATE TABLE articles (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    slug VARCHAR(255) UNIQUE NOT NULL,
    content TEXT NOT NULL,
    excerpt TEXT,
    featured_image VARCHAR(255),
    author_id INTEGER REFERENCES users(id),
    category_id INTEGER REFERENCES categories(id),
    status article_status DEFAULT 'draft',
    published_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Performance indexes
CREATE INDEX idx_articles_status_published ON articles(status, published_at DESC);
CREATE INDEX idx_articles_category ON articles(category_id);
CREATE INDEX idx_articles_author ON articles(author_id);
```
